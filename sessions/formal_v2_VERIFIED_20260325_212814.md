THEOREM: The composition of two polynomial-time computable functions is itself polynomial-time computable.

PROOF SKETCH:

2. By definition, there exist polynomials p_f and p_g such that for all inputs x, the computation time of f(x) is bounded by p_f(|x|) and the computation time of g(x) is bounded by p_g(|x|).
3. Consider the composition h(x) = f(g(x)).
4. The computation time of h(x) is the sum of the time to compute g(x) and the time to compute f(g(x)).
5. Since |g(x)| ≤ p_g(|x|) (as g is polynomial-time computable, it can only produce outputs of polynomial length), the time to compute f(g(x)) is bounded by p_f(p_g(|x|)).
6. Thus, the total computation time of h(x) is bounded by p_g(|x|) + p_f(p_g(|x|)), which is a polynomial in |x|.
7. Therefore, h is polynomial-time computable.

5. **Polynomial Time Verification**: Show that the number of decompositions checked is bounded by a polynomial in the length of w, ensuring the overall procedure runs in polynomial time.

import computability.language
open Computability

-- Polynomial-time decidability
def polyTimeDecidable (L : Language) : Prop :=
∃ (M : TM0), ∃ (p : ℕ → ℕ), (∀ (x : String), M.eval x = some (x ∈ L)) ∧ (∀ (x : String), M.eval x ≠ none) ∧ (∀ (n : ℕ), ∃ (k : ℕ), ∀ (x : String), x.length = n → M.steps x ≤ p n)

-- Kleene star operation
def kleeneStar (L : Language) : Language :=
{ w : String | ∃ (n : ℕ), ∃ (s : Fin n → String), (∀ (i : Fin n), s i ∈ L) ∧ (w = String.join (s ∘ Fin.val)) }

-- Theorem statement
theorem polyTimeDecidable_kleeneStar (L : Language) (h : polyTimeDecidable L) : polyTimeDecidable (kleeneStar L) :=
begin
  -- Unpack the hypothesis
  rcases h with ⟨M, p, hM, hM', hM''⟩

  -- Construct a new Turing machine
  let N : TM0 :=
  { -- Initialize variables
    step := fun (q : Config) =>
      match q.state with
      | 0 => -- Base case: empty string
        if q.input.isEmpty then
          { state := 1, input := q.input }
        else -- Recursive case: try all possible decompositions
          let u := q.input.take (q.input.length - 1)
          let v := q.input.drop 1
          if M.eval u = some true then
            { state := 0, input := v }
          else
            { state := 2, input := q.input }
      | 1 => -- Accept if the entire string is processed
        if q.input.isEmpty then
          { state := 3, input := q.input }
        else
          { state := 2, input := q.input }
      | _ => -- Reject if no decomposition is found
        { state := 3, input := q.input }
    , -- Accepting states
    accept := fun (q : Config) => q.state = 3
  }

  let q : ℕ → ℕ := fun (n : ℕ) => (n + 1) * p n

  apply Exists.intro N
  apply Exists.intro q
  apply And.intro

    apply Exists.intro fun (_ : Fin 0) => String.empty

  simp [N]
  cases x.length with
  | zero =>

    apply Nat.lt_of_mul_lt_mul_left
    apply Nat.lt_of_add_lt_add_left
