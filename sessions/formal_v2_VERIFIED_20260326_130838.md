import Mathlib
import computability.turing_machine
import data.polynomial

-- Define the polynomial-time decidable language structure
def poly_time_decidable (L : set (list Σ)) : Prop :=
  ∃ (M : turing_machine Σ bool),
    (∀ (x : list Σ), M.eval x = some (x ∈ L)) ∧
    ∃ (p : polynomial ℕ), ∀ (x : list Σ), M.time x ≤ p (x.length)

-- Define the concatenation operation on languages
def concat_lang (L₁ L₂ : set (list Σ)) : set (list Σ) :=
  { w | ∃ (u ∈ L₁) (v ∈ L₂), w = u ++ v }

theorem concat_poly_time_decidable {L₁ L₂ : set (list Σ)} (h₁ : poly_time_decidable L₁) (h₂ : poly_time_decidable L₂) :
  poly_time_decidable (concat_lang L₁ L₂) :=
begin
  -- Extract the deciders for L₁ and L₂
  obtain ⟨M₁, h₁_eval, p₁, h₁_time⟩ := h₁
  obtain ⟨M₂, h₂_eval, p₂, h₂_time⟩ := h₂

  -- Construct the decider for L₁ ◦ L₂

  { eval := λ w,
      let splits := list.split w in
      let eval_split (u : list Σ) (v : list Σ) :=
        if M₁.eval u = some true ∧ M₂.eval v = some true then some true else some false in
      list.foldl (λ b uv, if b = some true then some true else eval_split uv.1 uv.2) (some false) splits
    , time := λ w, (p₁ (w.length) + p₂ (w.length)) * w.length }

  -- Show that M decides L₁ ◦ L₂
  have h_eval : ∀ (w : list Σ), M.eval w = some (w ∈ concat_lang L₁ L₂),
  begin
    intro w,
    simp [M, concat_lang],
    apply list.induction_on w,
    { simp },
    { intros a w ih,
      simp [list.split],
      cases list.split w,
      { simp },
      { intros u v,
        simp [eval_split],
        cases M₁.eval u,
        { cases M₂.eval v,
          { simp },
          { simp } },
        { simp } } }
  end

  -- Show that M runs in polynomial time
  have h_time : ∃ (p : polynomial ℕ), ∀ (w : list Σ), M.time w ≤ p (w.length),

    intro w,
    simp [M],
    linarith
  end

  -- Conclude that L₁ ◦ L₂ is polynomial-time decidable
  exact ⟨M, h_eval, (p₁ + p₂) * polynomial.X, h_time⟩
