  ∃ M : TuringMachine Σ, M.compute f ∧ M.polyTime

theorem composition (f : Σ* → Σ*) (L : Language Σ) :
  polyTimeComputable f → polyTimeDecidable L → polyTimeDecidable {x | f x ∈ L} :=

              let ⟨y, hy, hx⟩ := Ml.halt x
