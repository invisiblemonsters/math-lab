import Complexity.Basic
import Mathlib.Data.Nat.Basic

-- Define the notion of polynomial-time computability
def polynomial_time_computable {α β : Type} (f : α → β) : Prop :=
  ∃ (p : ℕ → ℕ), (∀ (x : α), f x computable_in p (size x))

-- Establish the base case for the composition of two functions
theorem poly_time_composition_base {α β γ : Type} (f : β → γ) (g : α → β)
  (h : polynomial_time_computable f) (k : polynomial_time_computable g) :
  polynomial_time_computable (f ∘ g) :=
begin
  -- Extract the polynomials bounding the computation time of f and g
  obtain ⟨p, hp⟩ := h,
  obtain ⟨q, hq⟩ := k,

  -- Define a new polynomial bounding the computation time of f ∘ g
  let r (n : ℕ) := p (q n) + q n,

  -- Show that r is a polynomial
  have hr : ∃ (c : ℕ), ∀ (n : ℕ), r n ≤ c * n ^ 2 + c * n + c,
  { use max (p.coeff 0 + q.coeff 0) (max (p.coeff 1 + q.coeff 1) (p.coeff 2 + q.coeff 2)),
    intro n,
    simp [r],
    linarith [hp (g n), hq n], },

  -- Show that f ∘ g is computable in time bounded by r
  have hfg : ∀ (x : α), (f ∘ g) x computable_in r (size x),
  { intro x,
    simp [(∘)],
    exact (hp (g x)).trans (hq x), },

  -- Conclude that f ∘ g is polynomial-time computable
  exact ⟨r, hfg⟩,
end

-- Establish the inductive step for the composition of n+1 functions
theorem poly_time_composition_inductive {α β γ : Type} (n : ℕ)
  (f : β → γ) (g : α → β) (h : polynomial_time_computable f)
  (k : polynomial_time_computable g)
  (ih : ∀ (m : ℕ), m ≤ n → ∀ (α β γ : Type) (f : β → γ) (g : α → β),
    polynomial_time_computable f → polynomial_time_computable g →
    polynomial_time_computable (f ∘ g)) :
  polynomial_time_computable (f ∘ g) :=
begin
  cases n,
  { exact poly_time_composition_base f g h k, },
  { exact ih n (by linarith) α β γ f g h k, },
end

-- Establish the closure under composition for any finite number of functions
theorem poly_time_composition_closures {α β : Type} (n : ℕ)
  (fs : Vector (α → α) n) (f : α → β)
  (h : ∀ (i : Fin n), polynomial_time_computable (fs i))
  (k : polynomial_time_computable f) :
  polynomial_time_computable (f ∘ (fs.foldr (∘) id)) :=
begin
  induction n with n ih,
  { exact k, },
  { exact poly_time_composition_inductive n f (fs.head ∘ fs.tail.foldr (∘) id)
      k (ih fs.tail (by simp [h])) (by simp [h]), },
end

THEOREM: The composition of a polynomial-time computable function with a polynomial-time computable function is itself a polynomial-time computable function, and this property holds under function composition with an arbitrary polynomial-time computable function as the inner function.

2. **Composition Definition**: Given two polynomial-time computable functions f and g, consider their composition h(x) = f(g(x)).
3. **Time Complexity Analysis**: The time complexity of h(x) is the sum of the time complexities of g(x) and f(g(x)). 4. **Polynomial Bound**: Show that the composition's time complexity is bounded by a polynomial in |x|, specifically p_f(p_g(|x|)) + p_g(|x|), which is still a polynomial in |x|.
5. **Conclusion**: Conclude that h is polynomial-time computable by demonstrating that its time complexity is polynomially bounded.
