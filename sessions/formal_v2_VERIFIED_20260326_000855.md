THEOREM: The Cartesian product of two finite sets A and B is finite, and its cardinality is equal to the product of the cardinalities of A and B.

import Data.Set.Finite

theorem cardinality_product (A B : Set ℕ) (hA : A.Finite) (hB : B.Finite) : 
  (Set.prod A B).Finite ∧ (Set.prod A B).card = A.card * B.card :=
begin
  obtain ⟨n, ⟨fA, hfA⟩⟩ := hA
  obtain ⟨m, ⟨fB, hfB⟩⟩ := hB
  let f : ℕ → ℕ × ℕ := fun k => (fA (k / m), fB (k % m))
  have hf : ∀ (x : ℕ × ℕ), x ∈ Set.prod A B → ∃ k < n * m, f k = x :=
  begin
    intro x hx
    obtain ⟨hx1, hx2⟩ := Set.mem_prod.mp hx
    obtain ⟨k1, hk1⟩ := hfA hx1
    obtain ⟨k2, hk2⟩ := hfB hx2
    let k := m * k1 + k2
    have hk : k < n * m := 
    begin
      apply Nat.add_lt_of_lt_of_le
      exact hk2
      exact Nat.mul_le_mul_left _ hk1
    end
    use k, hk
    simp [f, Nat.div_add_mod, Nat.mod_add_mod]
  end
  have hf' : ∀ (k < n * m), (f k).1 ∈ A ∧ (f k).2 ∈ B :=
  begin
    intro k hk
    obtain ⟨k1, k2, hk12⟩ := (Nat.div_mod_unique k m).symm
    have hk1 : k1 < n := 
    begin
      apply Nat.lt_of_mul_lt_mul_left
      exact hk
      exact Nat.pos_of_ne_zero (NeZero.of_lt hk)
    end
    have hk2 : k2 < m := 
    begin
      apply Nat.mod_lt
      exact Nat.pos_of_ne_zero (NeZero.of_lt hk)
    end
    simp [f, hk12]
    exact ⟨hfA.2 hk1, hfB.2 hk2⟩
  end
  have h : Bijection f (Set.Iio (n * m)) (Set.prod A B) :=
  begin
    apply Bijection.mk
    apply Injective.mk
    intro k1 k2 h
    obtain ⟨k1', k2', hk12⟩ := (Nat.div_mod_unique k1 m).symm
    obtain ⟨k1'', k2'', hk12'⟩ := (Nat.div_mod_unique k2 m).symm
    simp [f, hk12, hk12'] at h
    apply Nat.div_mod_unique
    exact h.1
    exact h.2
    apply Surjective.mk
    intro x hx
    obtain ⟨k, hk⟩ := hf x hx
    use k
    exact hk.2
  end
  have hcard : (Set.prod A B).card = n * m := 
  begin
    apply Cardinal.mk_eq_of_bijection
    exact h
    exact ⟨h.finite_range, hcard⟩
  end
  exact ⟨h.finite_range, hcard⟩
end
