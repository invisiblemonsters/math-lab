import Mathlib
import Algebra.Polynomial

-- Define the notion of a polynomial function
def is_polynomial (f : ℕ → ℕ) : Prop :=
  ∃ (n : ℕ) (a : Fin (n + 1) → ℕ), ∀ (x : ℕ),
    f x = ∑ i in Fin.range (n + 1), a i * x ^ (n - i)

-- Define a helper lemma for polynomial arithmetic
lemma is_polynomial_of_form (n : ℕ) (a : Fin (n + 1) → ℕ) :
  is_polynomial (λ x, ∑ i in Fin.range (n + 1), a i * x ^ (n - i)) :=
by { use n, use a, intro x, rfl }

-- Define a helper lemma for the multiplication of polynomials
lemma is_polynomial_mul (p q : ℕ → ℕ) (hp : is_polynomial p) (hq : is_polynomial q) :
  is_polynomial (λ x, p x * q x) :=
begin
  cases hp with p_n p_a,
  cases hq with q_n q_a,
  use (p_n + q_n),
  use (λ i, ∑ j in Fin.range (i + 1), p_a j * q_a (i - j)),
  intro x,
  simp [Fin.sum_range],
  rw [Fin.sum_range, Fin.sum_range],
  simp [mul_comm, mul_assoc],
  apply Fin.sum_congr,
  intro i,
  simp [Fin.sum_range],
  apply Fin.sum_congr,
  intro j,
  simp [mul_comm, mul_assoc],
  omega
end

-- Define a helper lemma for the power of polynomials
lemma is_polynomial_pow (p : ℕ → ℕ) (hp : is_polynomial p) (n : ℕ) :
  is_polynomial (λ x, p x ^ n) :=
begin
  induction n with n ih,
  { use 0, use (λ _, 1), intro x, simp },
  { apply is_polynomial_mul, exact hp, exact ih }

  use (λ i, p_a i),
