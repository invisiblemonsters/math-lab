THEOREM: If there exists a polynomial-time reduction from language L1 to L2, and L2 is in P, then L1 is also in P.

import Mathlib
import Computability.PolynomialTime

structure Language where
  decider : String → Bool

class P (L : Language) where
  isDecidableInPolyTime : ∃ D, (∀ w, D w = L.decider w) ∧ IsPolyTime D

def Reduction (L1 L2 : Language) :=
  ∃ f, (∀ w, L1.decider w = L2.decider (f w)) ∧ IsPolyTime f

theorem composition_of_poly_time_functions {f g : String → String} (hf : IsPolyTime f) (hg : IsPolyTime g) :
    IsPolyTime (g ∘ f) :=
  IsPolyTime.composition hf hg

theorem reduction_preserves_poly_time (L1 L2 : Language) (h : Reduction L1 L2) (hL2 : P L2) : P L1 :=
  match h with
  | ⟨f, hf⟩ =>
    match hf with
    | ⟨hfw, hfpt⟩ =>
      match hL2.isDecidableInPolyTime with
      | ⟨D, hD⟩ =>
        P.mk
          ⟨D ∘ f, fun w => Eq.trans (hfw w) (hD.1 (f w)),
            composition_of_poly_time_functions hfpt hD.2⟩

theorem reduction_preserves_poly_time' (L1 L2 : Language)
    (h : ∃ f, (∀ w, L1.decider w = L2.decider (f w)) ∧ IsPolyTime f)
    (hL2 : ∃ D, (∀ w, D w = L2.decider w) ∧ IsPolyTime D) :
    ∃ D, (∀ w, D w = L1.decider w) ∧ IsPolyTime D :=

    match hf with
    | ⟨hfw, hfpt⟩ =>
