theorem reducible_concat {L M : set String} (h : L ≤ₚ M) : {uv : String | ∃ u v, uv = u ++ v ∧ u ∈ L ∧ v ∈ L} ≤ₚ M :=
begin
  cases' h with f hf,
  use (λ uv, f (uv.take (uv.length / 2)) ++ f (uv.drop (uv.length / 2))),
  have htake : poly_time (λ uv : String, uv.take (uv.length / 2)) :=
    by apply poly_time_take; apply poly_time_div; apply poly_time_length; apply poly_time_nat,
  have hdrop : poly_time (λ uv : String, uv.drop (uv.length / 2)) :=
    by apply poly_time_drop; apply poly_time_div; apply poly_time_length; apply poly_time_nat,
  have hcomp : poly_time (λ uv, f (uv.take (uv.length / 2))) :=
    by apply composition_poly_time; exact htake; exact hf,
  have hcomp' : poly_time (λ uv, f (uv.drop (uv.length / 2))) :=
    by apply composition_poly_time; exact hdrop; exact hf,
  have hconcat : poly_time (λ uv, f (uv.take (uv.length / 2)) ++ f (uv.drop (uv.length / 2))) :=
    by apply poly_time_append; exact hcomp; exact hcomp',
  exact hconcat,
  intro uv,
  constructor,
  · intro huv,
    cases' huv with u huv,
    cases' huv with v huv,
    cases' huv with huv huvL,
    cases' huvL with hu hv,
    simp [hf],
    apply And.intro,
    · apply hf.1; exact hu,
    · apply hf.1; exact hv,
  · intro huv,
    have := hf.2 huv,
    simp at this,
    cases' this with hu hv,
    use uv.take (uv.length / 2),
    use uv.drop (uv.length / 2),
    constructor,
    · simp,
    constructor,
    · apply hf.2; exact hu,
    · apply hf.2; exact hv,
end
