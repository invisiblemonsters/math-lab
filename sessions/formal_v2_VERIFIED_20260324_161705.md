5. **Decidability of h(L)**: Since M' can effectively decide membership in h(L) by exploring all possible decompositions (which are polynomially bounded if h maps symbols to strings of fixed length), h(L) is decidable.

import computability.turing_machine

-- Define the homomorphism on strings
def homomorphism (Σ Γ : Type) [Fintype Σ] [Fintype Γ] (h : Σ → List Γ) : List Σ → List Γ
| [] => []
| σ :: w => h σ ++ homomorphism h w

-- Theorem statement
theorem decidability_closed_under_homomorphism (Σ Γ : Type) [Fintype Σ] [Fintype Γ]
  (L : Set (List Σ)) (h : Σ → List Γ) (M : TuringMachine Σ Bool)
  (M_decides_L : ∀ w, M.eval w = (w ∈ L)) :
  ∃ M' : TuringMachine Γ Bool, ∀ v, M'.eval v = (v ∈ Set.image (homomorphism h) L) :=
begin
  -- Constructing M'
  let M' : TuringMachine Γ Bool := {
    next := fun (q : M.States) (v : Γ) =>
      match q with
      | M.start =>
        -- Nondeterministically guess a decomposition of v
        { q' := M.start, tape := v :: M.tape, result := M.result } ∨
        { q' := M.start, tape := [] :: M.tape, result := M.result }
      | M.loop σ =>
        -- Simulate M on the corresponding w
        match M.next (M.loop σ) σ with
        | { q' := M.halt b, .. } =>
          { q' := M.halt b, tape := M.tape, result := b }
        | { q' := q', .. } =>
          { q' := M.loop q', tape := M.tape, result := M.result }
      | M.halt b =>
        { q' := M.halt b, tape := M.tape, result := b }
    ,
    start := M.start,
    halt := fun q => q = M.halt true ∨ q = M.halt false,
    result := fun q => q = M.halt true
  }

  -- Verify the homomorphism property
  have homomorphism_property : ∀ w, homomorphism h w = List.bind w h :=
  begin
    intro w
    induction w
    case nil =>
      simp [homomorphism]
    case cons σ w ih =>
      simp [homomorphism, ih]

  intro v
  simp [M'.eval, homomorphism_property]
  apply propext
  split
  case h.mp =>
    intro h_v
    use List.bind v fun γ => List.pmap (fun σ => h σ) (List.pmap (fun γ' => γ') γ)
    simp [List.bind, List.pmap]
    apply List.exists_imp_exists
    intro w h_w
    simp [h_w]
    apply M_decides_L
  case h.mpr =>
    intro h_v
    apply List.exists_imp_exists
    intro w h_w
    simp [h_w]
    apply M_decides_L
