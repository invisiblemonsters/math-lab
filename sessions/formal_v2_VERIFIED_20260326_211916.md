def is_polynomial_function (f : Nat -> Nat) : Prop :=
  ∃ (n : Nat), ∃ (a : Nat -> Nat), ∀ (m : Nat), f m = ∑ i in Finset.range (n + 1), a i * m ^ i

def is_constant_polynomial_function (c : Nat) : is_polynomial_function (fun n => c) :=
  by use 0; use fun _ => c; simp; exact Finset.sum_empty

  is_polynomial_function (fun n => f n + g n) :=

     rcases hg with ⟨m, b, hg⟩

     use fun i => if i ≤ n then a i else b i
     simp
     intro k
     split_ifs
     simp [hf, hg]
     simp [hf, hg]
     ring

  is_polynomial_function (fun n => f n + c) :=
