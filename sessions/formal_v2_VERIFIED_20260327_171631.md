-- Define polynomial-time computable functions and relations
def polyTimeComputableFunction (f : Nat → Nat) : Prop :=
  ∃ (c : Nat) (n : Nat), ∀ (x : Nat), Complexity.Basic.T (f x) ≤ c * x ^ n + c

def polyTimeComputableRelation (R : Nat → Nat → Prop) : Prop :=
  ∃ (f : Nat → Nat → Bool), polyTimeComputableFunction (λ (x : Nat), f x) ∧
    ∀ (x y : Nat), R x y ↔ f x y = true

-- Theorem: Composition of a polynomial-time computable function with a polynomial-time computable relation
theorem compositionPolyTimeComputable (f g : Nat → Nat) (R : Nat → Nat → Prop)
  (hf : polyTimeComputableFunction f) (hg : polyTimeComputableFunction g)
  (hR : polyTimeComputableRelation R) :
  polyTimeComputableRelation (λ (x y : Nat), R (f x) (g y)) :=
begin
  -- Extract the algorithms and bounds for f, g, and R
  obtain ⟨c₁, n₁, hf⟩ := hf
  obtain ⟨c₂, n₂, hg⟩ := hg
  obtain ⟨fR, hfR, hR⟩ := hR

  -- Construct the algorithm for the composed relation
  let fS : Nat → Nat → Bool := λ (x y : Nat), fR (f x) (g y)

  -- Show that the constructed algorithm is polynomial-time
  have hfS : polyTimeComputableFunction (λ (x : Nat), fS x) :=
  begin
    apply Exists.intro (c₁ * c₂ + c₁ + c₂)
    apply Exists.intro (max n₁ n₂ + 1)
    intro x
    simp [fS]
    apply le_trans (hf x)
    apply le_trans (hg x)
    omega
  end

  -- Show that the constructed algorithm correctly decides the composed relation
  have hS : ∀ (x y : Nat), R (f x) (g y) ↔ fS x y = true :=
  begin
    intro x y
    apply hR
  end

  -- Conclude that the composed relation is polynomial-time computable
  apply Exists.intro fS
  apply And.intro hfS hS
end

THEOREM: The set of polynomial-time decidable languages is closed under the Kleene star operation. Formally, if a language L is in P, then the language L* = { w | w is a concatenation of finitely many strings from L } is also in P.

PROOF SKETCH:

2. **Decidability of L**: Since L is in P, there exists a polynomial-time Turing machine M that decides L.
3. **Constructing M' for L***: Design a polynomial-time Turing machine M' that, given input w, iteratively checks all possible decompositions of w into substrings from L.

LEAN HINT:
- Imports: `import complexity.basic`, `import data.list`

import Mathlib
import complexity.basic
import data.list

open complexity

theorem P_closed_under_Kleene_star {Σ : Type} [fintype Σ] (L : language Σ) :
  L ∈ P → L* ∈ P :=
begin
  rintro ⟨M, hM⟩,
  let M' : pTM Σ bool := 
  { 
    -- Initialize M' with the input string w
    init := λ w, (w, 0),
    -- Step function: for each state (w, i), try to decompose w into substrings from L
    step := λ (w, i), 
      if i < w.length then 
        let w₁ := w.take i in 
        let w₂ := w.drop i in 
        if M.eval w₁ = tt ∧ M.eval w₂ = tt then 
          (w₂, i + 1) 
        else 
          (w, i + 1) 
      else 
        (w, i),
    -- Halt when all decompositions have been checked
    halt := λ (w, i), i ≥ w.length,
    -- Accept if a valid decomposition is found
    accept := λ (w, i), ∃ j < w.length, M.eval (w.take j) = tt ∧ M.eval (w.drop j) = tt
  },
  -- Verify that M' runs in polynomial time
  have M'_time : M'.eval_time ≤ polynomial (λ n, n^2),
  { 
    induction w_length : w.length using nat.strong_induction_on with w ih,
    { simp },
    cases w,
    { simp },
    cases w,
    { simp },
    simp [M'.step],
    apply linarith,
    apply ih,
    apply nat.lt_pred_iff.mpr,
    apply w_length
  },
  -- Verify that M' correctly decides membership in L*
  have M'_correct : ∀ w, M'.eval w = tt ↔ w ∈ L*,

    intro w,
    split,
    { 
      rintro ⟨i, h⟩,
      cases h with j hj,
      cases hj with h₁ h₂,
      apply language.star.mem,
      use [w.take j, w.drop j],
      simp,
      apply hM.2,
      apply h₁,
      apply hM.2,
      apply h₂
    },
    { 
      intro hw,
      apply hw.cases_on,
      { simp },
      { 
        rintro ⟨w₁, w₂, hw₁, hw₂⟩,
        use w₁.length,
        simp,
        apply hM.2,
        apply hw₁,
        apply hM.2,
        apply hw₂
      }
    }
  },
  -- Conclude that L* ∈ P
  use M',
  split,
  apply M'_time,
  apply M'_correct
end
