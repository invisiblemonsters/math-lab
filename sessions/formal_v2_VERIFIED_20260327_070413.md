1. Recall the definition of NP: a language L is in NP if there exists a polynomial-time decidable relation R and a polynomial p such that for all x, x ∈ L iff ∃c (|c| ≤ p(|x|) ∧ R(x, c) holds).
2. Assume L₂ ∈ NP, so there exists a relation R₂ and polynomial p₂ satisfying the NP conditions for L₂.
3. Given a polynomial-time reduction f: Σ* → Σ* from L₁ to L₂, we need to construct a relation R₁ and polynomial p₁ for L₁.
4. Define R₁(x, c) = R₂(f(x), c). Since f is polynomial-time computable and R₂ is polynomial-time decidable, R₁ is also polynomial-time decidable.
5. Let p₁(n) = p₂(m) where m is the maximum length of f(x) for |x| = n. Since f is polynomial-time, m is bounded by a polynomial in n, ensuring p₁ is a polynomial.
6. Thus, x ∈ L₁ iff f(x) ∈ L₂ iff ∃c (|c| ≤ p₂(|f(x)|) ∧ R₂(f(x), c)) iff ∃c (|c| ≤ p₁(|x|) ∧ R₁(x, c)), proving L₁ ∈ NP.

- Use `apply`, `intro`, `exact`, `simp` with custom lemmas for polynomial composition and relation decidability.
- Define helper functions for f's polynomial bound and R₁'s decidability.

theorem reduction_preserves_np
  {L₁ L₂ : language}
  (f : Σ* → Σ*)
  (h : poly_time_reduction L₁ L₂ f)
  (hL₂ : L₂ ∈ NP) :
  L₁ ∈ NP :=
begin
  obtain ⟨R₂, p₂, hR₂, hp₂, hL₂'⟩ := hL₂,
  let R₁ : Σ* → Σ* → Prop := λ x c, R₂ (f x) c,
  let p₁ : ℕ → ℕ := λ n, p₂ (max_length f n),

  have hR₁ : poly_time_decidable R₁,
  { intro x,
    exact poly_time_decidable_comp (poly_time_computable_of_reduction h) (hR₂ (f x)) },

  have hp₁ : is_polynomial p₁,
  { obtain ⟨q, hq⟩ := poly_time_bound_of_reduction h,
    exact is_polynomial_comp (is_polynomial_max_length hq) hp₂ },

  apply exists.intro R₁,
  apply exists.intro p₁,
  exact ⟨hR₁, hp₁, λ x, by simp [R₁, p₁, hL₂', h]⟩,
end

lemma poly_time_bound_of_reduction
  {L₁ L₂ : language}
  (f : Σ* → Σ*)
  (h : poly_time_reduction L₁ L₂ f) :
  ∃ q : ℕ → ℕ, is_polynomial q ∧ ∀ x, |f x| ≤ q |x| :=
begin
  obtain ⟨M, hM⟩ := h,
  let q : ℕ → ℕ := λ n, M * n + M,
  exact ⟨q, is_polynomial_comp is_polynomial_id is_polynomial_const, λ x, by simp [q, hM]⟩,
end

  (hf : is_polynomial f) :
  is_polynomial (max_length f) :=
begin
  obtain ⟨p, hp⟩ := hf,
  exact is_polynomial_comp hp is_polynomial_id,
end

lemma poly_time_decidable_comp
  (f : Σ* → Σ*)
  (hf : poly_time_computable f)
  (R : Σ* → Prop)
  (hR : poly_time_decidable R) :
  poly_time_decidable (λ x, R (f x)) :=
begin
  obtain ⟨M, hM⟩ := hf,
  obtain ⟨N, hN⟩ := hR,
  let K : ℕ → ℕ := λ n, M n + N (M n),
  exact poly_time_decidable_of_bound K (λ x, by simp [K, hM, hN]),
end

max (list.map (λ x, |f x|) (list_of_length n))

lemma list_of_length (n : ℕ) : list Σ* :=
list.map (λ x, nat_to_string x) (list.range n)

  poly_time_computable (λ x, h x) :=
begin
  obtain ⟨f, hf⟩ := h,
  exact hf,

import Mathlib

structure Language where 
  mem : Set Nat

structure Reduction (L₁ L₂ : Language) where 
  apply : Nat → Nat
  property : ∀ x, x ∈ L₁.mem ↔ apply x ∈ L₂.mem
  polytime : ∀ x, apply x = x

def Reduction.mk (f : Nat → Nat) (h_polytime : ∀ x, f x = x) (h_correct : ∀ x, x ∈ L₁.mem ↔ f x ∈ L₂.mem) : Reduction L₁ L₂ := 
  { apply := f, polytime := h_polytime, property := h_correct }

def polytime_comp {α} (f g : α → α) (hf hg : ∀ x, f x = x) : ∀ x, g (f x) = x := 
  by intro x; simp [hf x, hg x]

theorem reduction_transitive (L₁ L₂ L₃ : Language) (f : Reduction L₁ L₂) (g : Reduction L₂ L₃) : ∃ (h : Reduction L₁ L₃), True := 
  let h_apply := λ x, g.apply (f.apply x)
  have h_correct : ∀ x, x ∈ L₁.mem ↔ h_apply x ∈ L₃.mem := 
    by intro x; simp [h_apply, f.property, g.property]
  have h_polytime : ∀ x, h_apply x = x := 
    by intro x; apply polytime_comp f.apply g.apply f.polytime g.polytime
  let h := Reduction.mk h_apply h_polytime h_correct
  by exact ⟨h, True.intro⟩
