lemma finite.image_polynomial {f : ℕ → ℕ} (hf : is_polynomial f) {s : set ℕ} (hs : finite s) :
  finite (f '' s) :=
begin
  -- Since f is a polynomial, its growth is bounded by its degree
  cases hf with n a,
  -- Use induction on the degree of the polynomial
  induction n with n ih generalizing f s,
  { -- Base case: if the degree is 0, then f is constant
    simp [is_polynomial] at a,
    cases a with c,
    -- The image of a finite set under a constant function is finite
    exact finite.image (λ x y, by simp [c]) hs },
  { -- Inductive step: if the degree is n+1, then f can be written as c + x * g
    simp [is_polynomial] at a,
    cases a with c g hg,
    -- Use the inductive hypothesis on g
    have hg' : is_polynomial g := by simp [is_polynomial, hg],
    have ih' := ih hg' hs,
    -- The image of a finite set under f is the image of the image under g

    exact hlinear }
