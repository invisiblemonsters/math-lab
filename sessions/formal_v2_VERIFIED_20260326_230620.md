PROOF SKETCH:

2. **Union Construction**: To show that L₁ ∪ L₂ is decidable, construct a Turing machine that, on input x, simulates the decision procedures for L₁ and L₂ in parallel.
3. **Simulation Argument**: Argue that this simulation can be done by interleaving the steps of the two machines. If either machine halts and accepts, the constructed machine halts and accepts. If both machines halt and reject, the constructed machine halts and rejects.
4. **Halting Guarantee**: Since L₁ and L₂ are decidable, both simulations are guaranteed to halt. Thus, the constructed machine for L₁ ∪ L₂ also halts on all inputs, ensuring decidability.

LEAN HINT:
- Imports: `import computability.turing_machine`
- Tactics: `apply`, `use` (to construct the Turing machine), `simp` (to verify the halting condition), `exact` (to reference decidability of L₁ and L₂)

-- Define decidability of a language
def decidable_language (L : set string) : Prop :=
  ∃ (M : turing_machine), ∀ (x : string), (x ∈ L → M.accepts x) ∧ (x ∉ L → M.rejects x)

-- Theorem statement
theorem decidable_union (L₁ L₂ : set string) (h₁ : decidable_language L₁) (h₂ : decidable_language L₂) :
  decidable_language (L₁ ∪ L₂) :=
begin
  -- Extract the Turing machines for L₁ and L₂
  obtain ⟨M₁, h₁⟩ := h₁
  obtain ⟨M₂, h₂⟩ := h₂

  -- Construct the Turing machine for L₁ ∪ L₂
  use (interleave M₁ M₂)

  -- Verify the halting condition
  intro x
  simp only [interleave_accepts_iff, interleave_rejects_iff]
  cases (x ∈ L₁ ∨ x ∈ L₂)

  -- Case 1: x ∈ L₁ ∨ x ∈ L₂
  case inl hx :
    cases hx
    case inl hx₁ :
      exact Or.inl (h₁.1 x hx₁)
    case inr hx₂ :
      exact Or.inr (h₂.1 x hx₂)

  -- Case 2: x ∉ L₁ ∪ L₂
  case inr hx :
    have hx₁ : x ∉ L₁ := by simp [hx]
    have hx₂ : x ∉ L₂ := by simp [hx]
    exact And.intro (h₁.2 x hx₁) (h₂.2 x hx₂)
