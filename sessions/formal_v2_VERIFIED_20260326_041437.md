import Mathlib
import data.nat.basic

theorem sum_of_multiples (m n : ℕ) : 
  (finset.range n).sum (λ i, m * (i + 1)) = m * n * (n + 1) / 2 :=
begin
  induction n with k hk,
  { simp [finset.range_zero, finset.sum_empty, nat.div_zero, mul_zero] },
  { simp [finset.range_succ, finset.sum_insert, nat.succ_eq_add_one, add_assoc],
    rw [hk, mul_add, add_comm (m * k * (k + 1) / 2) (m * (k + 1))],
    norm_num,
    ring }
end
