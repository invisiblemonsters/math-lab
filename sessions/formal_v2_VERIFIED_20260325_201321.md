- Imports: `import data.nat.basic`
- Tactics: `induction n with n ih; simp [nat.add_comm, nat.mul_div_cancel]; ring`

import Mathlib

theorem sum_odd_numbers (n : ℕ) : 
  (Finset.sum (Finset.range n) (λ k, 2 * k + 1)) = n ^ 2 :=
