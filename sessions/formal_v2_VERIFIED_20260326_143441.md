4. **Conclusion**: Hence, g ∘ f is polynomial-time computable.

begin

    by apply add_polynomial; apply mul_polynomial; apply const_polynomial; apply add_polynomial; exact hM₁.2; exact hM₂.2

      apply iff.intro
      . intro h

        use v
        exact h₁
        exact h₂
        exact h₃

        use u
        use v
        exact h₁
        exact h₂
        exact h₃
