    (∀ (x : string), M.accepts x = x ∈ L) ∧ 
    (∀ (x : string), T (M.run x) ≤ c * (string.length x) ^ n)

-- Theorem statement
theorem composition_preserves_poly_time_decidability 
  (f : string → string) (L : set string) 
  (h_f : poly_time_computable f) 
  (h_L : poly_time_decidable L) : 
  poly_time_decidable { w | f w ∈ L } := 
begin
  -- Extract polynomial-time computable function properties
  obtain ⟨c_f, n_f, h_f⟩ := h_f,
  -- Extract polynomial-time decidable language properties

  let M_comp : TuringMachine := 
    { run := λ w, 
      let f_w := f w in 
      M_L.run f_w,
      accepts := λ w, 
      let f_w := f w in 
      M_L.accepts f_w },
  
  -- Prove that the new Turing machine decides the composition
  have h_comp_accepts : ∀ (w : string), M_comp.accepts w = f w ∈ L := 
    λ w, by simp [M_comp.accepts, h_L_accepts],

      apply le_trans (h_L_time (f w)),

      apply le_of_pow_le_pow,

      exact nat.pow_le_pow_of_le_left (le_max_right n_f n_L),
