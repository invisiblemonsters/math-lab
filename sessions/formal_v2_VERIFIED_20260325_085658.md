-- Define polynomial functions
def is_polynomial (f : ℕ → ℕ) : Prop :=
∃ p : ℕ → ℕ, (∀ n, f n ≤ p n) ∧
             (∃ k : ℕ, ∀ n, p n ≤ n ^ k)

-- Composition of polynomial functions is polynomial
theorem composition_polynomial (f g : ℕ → ℕ) :
  is_polynomial f → is_polynomial g → is_polynomial (f ∘ g) :=
begin
  rintros ⟨p_f, hf, k_f, hf_bound⟩ ⟨p_g, hg, k_g, hg_bound⟩,
  use (λ n, p_f (p_g n)),

    apply le_trans (hf (g n)),
    apply hf },

    intro n,
    apply le_trans (hf_bound (p_g n)),
    apply pow_mul_le_pow,
    apply hg_bound }
end

  is_polynomial (λ x, a * x ^ 2 + b * x + c) :=

  use (λ x, a * x ^ 2 + b * x + c),
  split,
  { intro n, refl },
  { use 2, intro n, simp, ring }
